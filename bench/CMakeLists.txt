    pthread
    libifm_nettle
    libifm_libgcrypt
    libifm_libxcrypt
    gpg-error

    PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/xcrypt
    PUBLIC ${CMAKE_SOURCE_DIR}/src
    PUBLIC ${CMAKE_SOURCE_DIR}/src/gcrypt
    PUBLIC ${CMAKE_SOURCE_DIR}/src/xcrypt
    )

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g")
